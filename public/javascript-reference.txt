   function lineDraw()
   {
      if(clicks % 2 == 0)
      {
      x2 = event.clientX - rect.left
      y2 = event.clientY - rect.top
      }
      else {
         x1 = event.clientX - rect.left
         y1 = event.clientY - rect.top
         
      }
      ctx.beginPath()
      ctx.moveTo(x1, y1)
      ctx.lineTo(x2, y2)
      ctx.stroke()
      clicks ++

   }
